import React, { useState, useEffect } from 'react';
import { motion } from 'framer-motion';
import { Sparkles, AlertTriangle, TrendingUp, Lightbulb, Loader2, Lock } from 'lucide-react';
import { base44 } from '@/api/base44Client';
import { Button } from '@/components/ui/button';
import { cn } from '@/lib/utils';

export default function AIInsights({ expenses, isPremium, onUpgrade }) {
  const [insights, setInsights] = useState(null);
  const [isLoading, setIsLoading] = useState(false);

  useEffect(() => {
    if (isPremium && expenses.length >= 5) {
      generateInsights();
    }
  }, [expenses.length, isPremium]);

  const generateInsights = async () => {
    if (!isPremium || expenses.length < 5) return;
    
    setIsLoading(true);
    try {
      const expenseSummary = expenses.slice(0, 50).map(e => ({
        description: e.description,
        amount: e.amount,
        category: e.category,
        date: e.date,
        merchant: e.merchant
      }));

      const result = await base44.integrations.Core.InvokeLLM({
        prompt: `Analyze these expenses and provide smart insights:
        
${JSON.stringify(expenseSummary, null, 2)}

Provide 3 actionable insights about:
1. Spending patterns or unusual expenses
2. Budget recommendations
3. Potential savings opportunities

Be concise and specific to the actual data.`,
        response_json_schema: {
          type: 'object',
          properties: {
            insights: {
              type: 'array',
              items: {
                type: 'object',
                properties: {
                  type: { type: 'string', enum: ['pattern', 'recommendation', 'alert'] },
                  title: { type: 'string' },
                  description: { type: 'string' }
                }
              }
            }
          }
        }
      });

      setInsights(result.insights);
    } catch (error) {
      console.error('Failed to generate insights:', error);
    }
    setIsLoading(false);
  };

  const iconMap = {
    pattern: TrendingUp,
    recommendation: Lightbulb,
    alert: AlertTriangle
  };

  const colorMap = {
    pattern: 'text-blue-600 bg-blue-50',
    recommendation: 'text-emerald-600 bg-emerald-50',
    alert: 'text-amber-600 bg-amber-50'
  };

  if (!isPremium) {
    return (
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ delay: 0.3 }}
        className="bg-gradient-to-br from-blue-50 to-indigo-50 rounded-lg p-5 border border-blue-100"
      >
        <div className="flex items-start gap-3">
          <div className="p-2 rounded-lg bg-blue-100">
            <Lock className="w-5 h-5 text-blue-600" />
          </div>
          <div className="flex-1">
            <h3 className="text-base font-semibold text-gray-900 flex items-center gap-2">
              AI Insights
              <span className="px-2 py-0.5 text-xs font-medium bg-gradient-to-r from-blue-500 to-indigo-600 text-white rounded-full">
                Premium
              </span>
            </h3>
            <p className="text-sm text-gray-600 mt-1">
              Get personalized spending analysis, budget recommendations, and unusual expense alerts.
            </p>
            <Button
              onClick={onUpgrade}
              className="mt-4 bg-gradient-to-r from-blue-500 to-indigo-600 hover:from-blue-600 hover:to-indigo-700 text-white rounded-lg"
            >
              Upgrade to Premium
            </Button>
          </div>
        </div>
      </motion.div>
    );
  }

  if (isLoading) {
    return (
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        className="bg-white rounded-lg p-5 border border-gray-200 shadow-sm"
      >
        <div className="flex items-center gap-3">
          <div className="p-2 rounded-lg bg-blue-100">
            <Sparkles className="w-5 h-5 text-blue-600" />
          </div>
          <h3 className="text-base font-semibold text-gray-900">AI Insights</h3>
        </div>
        <div className="flex items-center justify-center py-8">
          <Loader2 className="w-6 h-6 text-blue-500 animate-spin" />
          <span className="ml-2 text-sm text-gray-500">Analyzing your spending...</span>
        </div>
      </motion.div>
    );
  }

  if (!insights || insights.length === 0) {
    return (
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        className="bg-white rounded-lg p-5 border border-gray-200 shadow-sm"
      >
        <div className="flex items-center gap-3">
          <div className="p-2 rounded-lg bg-blue-100">
            <Sparkles className="w-5 h-5 text-blue-600" />
          </div>
          <h3 className="text-base font-semibold text-gray-900">AI Insights</h3>
        </div>
        <p className="text-sm text-gray-500 mt-4 text-center py-4">
          Add at least 5 expenses to get personalized insights
        </p>
      </motion.div>
    );
  }

  return (
    <motion.div
      initial={{ opacity: 0, y: 20 }}
      animate={{ opacity: 1, y: 0 }}
      transition={{ delay: 0.3 }}
      className="bg-white rounded-lg p-5 border border-gray-200 shadow-sm"
    >
      <div className="flex items-center gap-3 mb-4">
        <div className="p-2 rounded-lg bg-blue-100">
          <Sparkles className="w-5 h-5 text-blue-600" />
        </div>
        <h3 className="text-base font-semibold text-gray-900">AI Insights</h3>
      </div>
      
      <div className="space-y-3">
        {insights.map((insight, index) => {
          const Icon = iconMap[insight.type] || Lightbulb;
          return (
            <motion.div
              key={index}
              initial={{ opacity: 0, x: -20 }}
              animate={{ opacity: 1, x: 0 }}
              transition={{ delay: 0.1 * index }}
              className="p-3 bg-gray-50 rounded-lg"
            >
              <div className="flex items-start gap-3">
                <div className={cn("p-2 rounded-lg", colorMap[insight.type])}>
                  <Icon className="w-4 h-4" />
                </div>
                <div>
                  <h4 className="font-medium text-gray-900 text-sm">{insight.title}</h4>
                  <p className="text-xs text-gray-600 mt-1">{insight.description}</p>
                </div>
              </div>
            </motion.div>
          );
        })}
      </div>
    </motion.div>
  );
}
